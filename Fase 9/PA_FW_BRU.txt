# Palo Alto PA-440 CLI config equivalent voor Cisco FW_BRU (Fase 9)
# Aanpassen van poortnummers indien je andere fysieke interfaces gebruikt.
configure

# Virtual router
set network virtual-router VR1

# Zones
set zone Inside1 network layer3 ethernet1/1
set zone Inside2 network layer3 ethernet1/2
set zone WAN     network layer3 ethernet1/3
set zone P2P     network layer3 ethernet1/4
set zone Loopbacks network layer3 loopback.1

# Interfaces (pas poortnummers aan indien nodig)
set network interface ethernet ethernet1/1 layer3 ip 10.4.5.241/30
set network interface ethernet ethernet1/1 layer3 interface-management-profile ping-allow
set network interface ethernet ethernet1/1 layer3 virtual-router VR1
set network interface ethernet ethernet1/1 layer3 zone Inside1

set network interface ethernet ethernet1/2 layer3 ip 10.4.5.245/30
set network interface ethernet ethernet1/2 layer3 interface-management-profile ping-allow
set network interface ethernet ethernet1/2 layer3 virtual-router VR1
set network interface ethernet ethernet1/2 layer3 zone Inside2

set network interface ethernet ethernet1/3 layer3 ip 192.168.10.6/30
set network interface ethernet ethernet1/3 layer3 interface-management-profile ping-allow
set network interface ethernet ethernet1/3 layer3 virtual-router VR1
set network interface ethernet ethernet1/3 layer3 zone WAN

set network interface ethernet ethernet1/4 layer3 ip 172.16.0.13/30
set network interface ethernet ethernet1/4 layer3 interface-management-profile ping-allow
set network interface ethernet ethernet1/4 layer3 virtual-router VR1
set network interface ethernet ethernet1/4 layer3 zone P2P

# Loopback
set network interface loopback loopback.1 layer3 ip 10.4.5.255/32
set network interface loopback loopback.1 layer3 virtual-router VR1
set network interface loopback loopback.1 layer3 zone Loopbacks

# Static routes (primair via ethernet1/1, backup via ethernet1/2, default via WAN)
set network virtual-router VR1 routing-table ip static-route NET-10.4.0.0/22 interface ethernet1/1 destination 10.4.0.0/22 nexthop ip-address 10.4.5.242
set network virtual-router VR1 routing-table ip static-route NET-10.4.4.0/24 interface ethernet1/1 destination 10.4.4.0/24 nexthop ip-address 10.4.5.242
set network virtual-router VR1 routing-table ip static-route NET-10.4.5.0/26 interface ethernet1/1 destination 10.4.5.0/26 nexthop ip-address 10.4.5.242

set network virtual-router VR1 routing-table ip static-route NET-10.4.0.0/22-BK metric 5 interface ethernet1/2 destination 10.4.0.0/22 nexthop ip-address 10.4.5.246
set network virtual-router VR1 routing-table ip static-route NET-10.4.4.0/24-BK metric 5 interface ethernet1/2 destination 10.4.4.0/24 nexthop ip-address 10.4.5.246
set network virtual-router VR1 routing-table ip static-route NET-10.4.5.0/26-BK metric 5 interface ethernet1/2 destination 10.4.5.0/26 nexthop ip-address 10.4.5.246

set network virtual-router VR1 routing-table ip static-route DEFAULT interface ethernet1/3 destination 0.0.0.0/0 nexthop ip-address 192.168.10.5

# OSPF area 0 (zoals Cisco)
set network virtual-router VR1 protocol ospf enable yes
set network virtual-router VR1 protocol ospf router-id 10.4.5.255
set network virtual-router VR1 protocol ospf area 0.0.0.0
set network virtual-router VR1 protocol ospf area 0.0.0.0 interface ethernet1/1 enable yes
set network virtual-router VR1 protocol ospf area 0.0.0.0 interface ethernet1/2 enable yes
set network virtual-router VR1 protocol ospf area 0.0.0.0 interface ethernet1/3 enable yes
set network virtual-router VR1 protocol ospf area 0.0.0.0 interface ethernet1/4 enable yes
set network virtual-router VR1 protocol ospf area 0.0.0.0 interface loopback.1 enable yes passive yes

# Management-profiel voor ping
set network profiles interface-management-profile ping-allow ping yes

# Security policies (tijdelijk permissief; verstrengen naar behoefte)
set rulebase security rules TEMP-Inside-to-WAN from [ Inside1 Inside2 ] to [ WAN ] source any destination any application any service any action allow
set rulebase security rules TEMP-Inside-to-P2P from [ Inside1 Inside2 ] to [ P2P ] source any destination any application any service any action allow
set rulebase security rules TEMP-P2P-to-Inside from [ P2P ] to [ Inside1 Inside2 ] source any destination any application any service any action allow
set rulebase security rules TEMP-WAN-to-Inside from [ WAN ] to [ Inside1 Inside2 ] source any destination any application any service any action allow

# (Optioneel) NAT voor internet-uit via WAN:

commit
